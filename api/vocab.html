    Vocab

    local vocab = model:vocab()
local tokens, n = vocab:tokenize("Hello, world!", true, true)
local text = vocab:detokenize(tokens, n)
print(text)  --> "Hello, world!"
    _ptr cdata Opaque llama_vocab* (not owned, do not free).
    _lib cdata libllama.so namespace.
    _model cdata llama_model* for chat template calls.
    _bridge cdata ion7_bridge.so namespace.
    _ffi cdata LuaJIT ffi namespace.
    _tmpls cdata ion7_chat_templates_t* (owned, freed via ffi.gc).
    _piece_buf cdata Scratch buffer for token-to-piece conversion.
    _piece_big cdata Pre-alloc fallback buffer for oversized pieces.
    _piece_cache table Strong cache: token id → piece string.
    _tmpl_buf cdata Scratch buffer for chat template output.
    _dtok_buf cdata Scratch buffer for detokenize output.
